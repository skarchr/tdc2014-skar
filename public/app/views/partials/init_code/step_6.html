(function(){
            'use strict';
            angular.module('app').config(['$routeProvider', '$locationProvider', function($routeProvider, $locationProvider) {

                $locationProvider.html5Mode(true);

                $routeProvider.
                    when('/acme', {
                    templateUrl: '/path/acme'
                    controller: 'acmeController'
                }).
                    otherwise({
                    redirectTo: '/path/index'
                });
            }]);
        })();
