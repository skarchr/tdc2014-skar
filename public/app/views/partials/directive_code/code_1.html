(function(){
            'use strict';

            angular.module('app').controller('directiveController', ['$scope', '$timeout', function($scope, $timeout){
                $scope.run = false;
                $scope.jump = false;
                $scope.slide = false;

                $scope.ngJump = function(){

                    $scope.jump = true;

                    $timeout(function(){
                        $scope.jump = false;
                    }, 300);
                };

                $scope.ngSlide = function(){

                    $scope.slide = true;

                    $timeout(function(){
                        $scope.slide = false;
                    }, 300);
                };
            }]);
        })();
    

    

        <div>
            <button ng-click="ngJump()">Jump</button>
            <button ng-click="run = false">Stop</button>
            <button ng-click="run = !run" ng-disabled="run" class="btn btn-deafult {{run ? 'active':''}}">Run</button>
            <button ng-click="ngSlide()">Slide</button>
        </div>

        <div style="padding-bottom: {{jump ? '50px':'0'}}; padding-top: {{jump ? '0':'50px'}}"">
            <img src="/app/images/standing_blinking_512.gif" ng-hide="run || jump || slide"/>
            <img src="/app/images/running_512.gif" ng-show="run && !jump && !slide"/>
            <img src="/app/images/jump.png" ng-show="jump"/>
            <img src="/app/images/slide.png" ng-show="slide"/>
        </div>
