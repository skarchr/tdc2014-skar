$scope.watchSelect = "";
        $scope.watchNew = "";
        $scope.watchOld = "";

        $scope.$watch('watchSelect', function(newVal, oldVal){
            if(newVal !== oldVal){
                $scope.watchNew = newVal;
                $scope.watchOld = oldVal;
            }
        });

    

    

        <select ng-model="watchSelect" ng-options="item.name for item in repeat.module">
            <option>{{item.name}}</option>
        </select>

        <div>
            <h1>New:</h1>
            <img ng-src="{{watchNew.image}}"/>
            <h2>{{watchNew.name}}</h2>
            <h3>{{watchNew.number}}</h3>
        </div>

        <div>
            <h3 ng-show="watchOld !== ''">{{watchNew.number}} + {{watchOld.number}} = {{watchNew.number + watchOld.number}}</h3>
        </div>

        <div>
            <h1>Previous:</h1>
            <img ng-src="{{watchOld.image}}"/>
            <h2>{{watchOld.name}}</h2>
            <h3>{{watchOld.number}}</h3>
        </div>
